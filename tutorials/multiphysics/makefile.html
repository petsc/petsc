runp1:
	-@${MPIEXEC} -n 1 ./p1 -snes_monitor_short > p1_1.tmp 2>&1;   \
	   if (${DIFF} output/p1_1.out p1_1.tmp) then true; \
	   else echo "Possible problem with p1_1, diffs above"; fi; \
	   ${RM} -f p1_1.tmp
